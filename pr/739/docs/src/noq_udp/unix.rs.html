27/// Tokio-compatible UDP socket with some useful specializations
28///
29/// Unlike a standard tokio UDP socket, this allows ECN bits to be read and written on some
30/// platforms.
31#[derive(Debug)]
32pub struct UdpSocketState {
33    last_send_error: Mutex<Instant>,
34    max_gso_segments: AtomicUsize,
35    gro_segments: NonZeroUsize,
36    may_fragment: bool,

38    /// True if we have received EINVAL error from `sendmsg` system call at least once.
39    ///
40    /// If enabled, we assume that old kernel is used and switch to fallback mode.
41    /// In particular, we do not use IP_TOS cmsg_type in this case,
42    /// which is not supported on Linux <3.13 and results in not sending the UDP packet at all.
43    sendmsg_einval: AtomicBool,
44
45    /// Whether to use Apple's fast `sendmsg_x`/`recvmsg_x` APIs.
46    ///
47    /// These private APIs provide better performance but may not be available on all
48    /// Apple OS versions. Callers must verify availability before enabling.
49    #[cfg(apple_fast)]
50    apple_fast_path: AtomicBool,

53impl UdpSocketState {
54    pub fn new(sock: UdpSockRef<'_>) -> io::Result<Self> {
55        let io = sock.0;
56        let mut cmsg_platform_space = 0;
57        #[cfg(not(target_os = "redox"))]
58        if cfg!(target_os = "linux")
59            || cfg!(bsd)
60            || cfg!(apple)
61            || cfg!(target_os = "android")
62            || cfg!(solarish)
63        {
64            cmsg_platform_space +=
65                unsafe { libc::CMSG_SPACE(mem::size_of::<libc::in6_pktinfo>() as _) as usize };
66        }
67
68        assert!(
69            cmsg::LEN
70                >= unsafe { libc::CMSG_SPACE(mem::size_of::<libc::c_int>() as _) as usize }
71                    + cmsg_platform_space
72        );
73        assert!(
74            mem::align_of::<libc::cmsghdr>() <= mem::align_of::<cmsg::Aligned<[u8; 0]>>(),
75            "control message buffers will be misaligned"
76        );
77
78        io.set_nonblocking(true)?;
79
80        let addr = io.local_addr()?;
81        let is_ipv4 = addr.family() == libc::AF_INET as libc::sa_family_t;
82
83        // mac and ios do not support IP_RECVTOS on dual-stack sockets :(
84        // older macos versions also don't have the flag and will error out if we don't ignore it
85        #[cfg(not(any(
86            target_os = "openbsd",
87            target_os = "netbsd",
88            target_os = "dragonfly",
89            solarish
90        )))]
91        if (is_ipv4 || !io.only_v6()?)
92            && let Err(_err) =
93                set_socket_option(&*io, libc::IPPROTO_IP, libc::IP_RECVTOS, OPTION_ON)
94        {
95            crate::log::debug!("Ignoring error setting IP_RECVTOS on socket: {_err:?}");
96        }
97
98        let mut may_fragment = false;
99        #[cfg_attr(
100            not(any(target_os = "linux", target_os = "android")),
101            expect(unused_mut)
102        )]
103        let mut gro_segments = NonZeroUsize::MIN;
104
105        #[cfg(any(target_os = "linux", target_os = "android"))]
106        {
107            // Forbid IPv4 fragmentation. Set even for IPv6 to account for IPv6 mapped IPv4 addresses.
108            // Set `may_fragment` to `true` if this option is not supported on the platform.
109            may_fragment |= !set_socket_option_supported(
110                &*io,
111                libc::IPPROTO_IP,
112                libc::IP_MTU_DISCOVER,
113                libc::IP_PMTUDISC_PROBE,
114            )?;
115
116            if is_ipv4 {
117                set_socket_option(&*io, libc::IPPROTO_IP, libc::IP_PKTINFO, OPTION_ON)?;
118            } else {
119                // Set `may_fragment` to `true` if this option is not supported on the platform.
120                may_fragment |= !set_socket_option_supported(
121                    &*io,
122                    libc::IPPROTO_IPV6,
123                    libc::IPV6_MTU_DISCOVER,
124                    libc::IPV6_PMTUDISC_PROBE,
125                )?;
126            }
127
128            if set_socket_option(&*io, libc::SOL_UDP, libc::UDP_GRO, OPTION_ON).is_ok() {
129                // As defined in net/ipv4/udp_offload.c
130                // #define UDP_GRO_CNT_MAX 64
131                //
132                // NOTE: this MUST be set to UDP_GRO_CNT_MAX to ensure that the receive buffer size
133                // (get_max_udp_payload_size() * gro_segments()) is large enough to hold the largest GRO
134                // list the kernel might potentially produce. See
135                // https://github.com/quinn-rs/quinn/pull/1354.
136                gro_segments = NonZeroUsize::new(64).expect("known");
137            }
138        }
139        #[cfg(any(target_os = "freebsd", apple))]
140        {
141            if is_ipv4 {
142                // Set `may_fragment` to `true` if this option is not supported on the platform.
143                may_fragment |= !set_socket_option_supported(
144                    &*io,
145                    libc::IPPROTO_IP,
146                    libc::IP_DONTFRAG,
147                    OPTION_ON,
148                )?;
149            }
150        }
151        #[cfg(any(bsd, apple, solarish))]
152        // IP_RECVDSTADDR == IP_SENDSRCADDR on FreeBSD
153        // macOS uses only IP_RECVDSTADDR, no IP_SENDSRCADDR on macOS (the same on Solaris)
154        // macOS also supports IP_PKTINFO
155        {
156            if is_ipv4 {
157                set_socket_option(&*io, libc::IPPROTO_IP, libc::IP_RECVDSTADDR, OPTION_ON)?;
158            }
159        }
160
161        // Options standardized in RFC 3542
162        #[cfg(not(target_os = "redox"))]
163        if !is_ipv4 {
164            set_socket_option(&*io, libc::IPPROTO_IPV6, libc::IPV6_RECVPKTINFO, OPTION_ON)?;
165            set_socket_option(&*io, libc::IPPROTO_IPV6, libc::IPV6_RECVTCLASS, OPTION_ON)?;
166            // Linux's IP_PMTUDISC_PROBE allows us to operate under interface MTU rather than the
167            // kernel's path MTU guess, but actually disabling fragmentation requires this too. See
168            // __ip6_append_data in ip6_output.c.
169            // Set `may_fragment` to `true` if this option is not supported on the platform.
170            may_fragment |= !set_socket_option_supported(
171                &*io,
172                libc::IPPROTO_IPV6,
173                libc::IPV6_DONTFRAG,
174                OPTION_ON,
175            )?;
176        }
177
178        let now = Instant::now();
179        Ok(Self {
180            last_send_error: Mutex::new(now.checked_sub(2 * IO_ERROR_LOG_INTERVAL).unwrap_or(now)),
181            max_gso_segments: AtomicUsize::new(gso::max_gso_segments(&*io)),
182            gro_segments,
183            may_fragment,
184            sendmsg_einval: AtomicBool::new(false),
185            #[cfg(apple_fast)]
186            apple_fast_path: AtomicBool::new(false),
187        })
188    }

190    /// Sends a [`Transmit`] on the given socket
191    ///
192    /// This function will only ever return errors of kind [`io::ErrorKind::WouldBlock`].
193    /// All other errors will be logged and converted to `Ok`.
194    ///
195    /// UDP transmission errors are considered non-fatal because higher-level protocols must
196    /// employ retransmits and timeouts anyway in order to deal with UDP's unreliable nature.
197    /// Thus, logging is most likely the only thing you can do with these errors.
198    ///
199    /// If you would like to handle these errors yourself, use [`UdpSocketState::try_send`]
200    /// instead.
201    pub fn send(&self, socket: UdpSockRef<'_>, transmit: &Transmit<'_>) -> io::Result<()> {
202        match send(self, socket.0, transmit) {
203            Ok(()) => Ok(()),
204            Err(e) if e.kind() == io::ErrorKind::WouldBlock => Err(e),
205            // - EMSGSIZE is expected for MTU probes. Future work might be able to avoid
206            //   these by automatically clamping the MTUD upper bound to the interface MTU.
207            Err(e) if e.raw_os_error() == Some(libc::EMSGSIZE) => Ok(()),
208            Err(e) => {
209                log_sendmsg_error(&self.last_send_error, e, transmit);
210
211                Ok(())
212            }
213        }
214    }

378#[cfg(not(any(apple, target_os = "openbsd", target_os = "netbsd")))]
379fn send(
380    #[allow(unused_variables)] // only used on Linux
381    state: &UdpSocketState,
382    io: SockRef<'_>,
383    transmit: &Transmit<'_>,
384) -> io::Result<()> {
385    #[allow(unused_mut)] // only mutable on FreeBSD
386    let mut encode_src_ip = true;
387    #[cfg(target_os = "freebsd")]
388    {
389        let addr = io.local_addr()?;
390        let is_ipv4 = addr.family() == libc::AF_INET as libc::sa_family_t;
391        if is_ipv4 {
392            if let Some(socket) = addr.as_socket_ipv4() {
393                encode_src_ip = socket.ip() == &Ipv4Addr::UNSPECIFIED;
394            }
395        }
396    }
397    let mut msg_hdr: libc::msghdr = unsafe { mem::zeroed() };
398    let mut iovec: libc::iovec = unsafe { mem::zeroed() };
399    let mut cmsgs = cmsg::Aligned([0u8; cmsg::LEN]);
400    let dst_addr = socket2::SockAddr::from(transmit.destination);
401    prepare_msg(
402        transmit,
403        &dst_addr,
404        &mut msg_hdr,
405        &mut iovec,
406        &mut cmsgs,
407        encode_src_ip,
408        state.sendmsg_einval(),
409    );
410
411    loop {
412        let n = unsafe { libc::sendmsg(io.as_raw_fd(), &msg_hdr, 0) };
413
414        if n >= 0 {
415            return Ok(());
416        }
417
418        let e = io::Error::last_os_error();
419        match e.kind() {
420            // Retry the transmission
421            io::ErrorKind::Interrupted => continue,
422            io::ErrorKind::WouldBlock => return Err(e),
423            _ => {
424                // Some network adapters and drivers do not support GSO. Unfortunately, Linux
425                // offers no easy way for us to detect this short of an EIO or sometimes EINVAL
426                // when we try to actually send datagrams using it.
427                #[cfg(any(target_os = "linux", target_os = "android"))]
428                if let Some(libc::EIO) | Some(libc::EINVAL) = e.raw_os_error() {
429                    // Prevent new transmits from being scheduled using GSO. Existing GSO transmits
430                    // may already be in the pipeline, so we need to tolerate additional failures.
431                    if state.max_gso_segments().get() > 1 {
432                        crate::log::info!(
433                            "`libc::sendmsg` failed with {e}; halting segmentation offload"
434                        );
435                        state
436                            .max_gso_segments
437                            .store(1, std::sync::atomic::Ordering::Relaxed);
438                    }
439                }
440
441                // Some arguments to `sendmsg` are not supported. Switch to
442                // fallback mode and retry if we haven't already.
443                if e.raw_os_error() == Some(libc::EINVAL) && !state.sendmsg_einval() {
444                    state.set_sendmsg_einval();
445                    prepare_msg(
446                        transmit,
447                        &dst_addr,
448                        &mut msg_hdr,
449                        &mut iovec,
450                        &mut cmsgs,
451                        encode_src_ip,
452                        state.sendmsg_einval(),
453                    );
454                    continue;
455                }
456
457                return Err(e);
458            }
459        }
460    }
461}

470pub(crate) fn send_single(
471    state: &UdpSocketState,
472    io: SockRef<'_>,
473    transmit: &Transmit<'_>,
474) -> io::Result<()> {
475    let mut hdr: libc::msghdr = unsafe { mem::zeroed() };
476    let mut iov: libc::iovec = unsafe { mem::zeroed() };
477    let mut ctrl = cmsg::Aligned([0u8; cmsg::LEN]);
478    let addr = socket2::SockAddr::from(transmit.destination);
479    prepare_msg(
480        transmit,
481        &addr,

489    Ok(())

492/// Receive using the batched `recvmmsg` syscall
493#[cfg(not(any(
494    apple,
495    target_os = "openbsd",
496    target_os = "netbsd",
497    target_os = "dragonfly",
498    target_os = "redox",
499    solarish
500)))]
501fn recv_via_recvmmsg(
502    io: SockRef<'_>,
503    bufs: &mut [IoSliceMut<'_>],
504    meta: &mut [RecvMeta],
505) -> io::Result<usize> {
506    let mut names = [MaybeUninit::<libc::sockaddr_storage>::uninit(); BATCH_SIZE];
507    let mut ctrls = [cmsg::Aligned(MaybeUninit::<[u8; cmsg::LEN]>::uninit()); BATCH_SIZE];
508    let mut hdrs = unsafe { mem::zeroed::<[libc::mmsghdr; BATCH_SIZE]>() };
509    let max_msg_count = bufs.len().min(BATCH_SIZE);
510    for i in 0..max_msg_count {
511        prepare_recv(
512            &mut bufs[i],
513            &mut names[i],
514            &mut ctrls[i],
515            &mut hdrs[i].msg_hdr,
516        );
517    }
518    let msg_count = retry_if_interrupted(|| unsafe {
519        libc::recvmmsg(
520            io.as_raw_fd(),
521            hdrs.as_mut_ptr(),
522            bufs.len().min(BATCH_SIZE) as _,
523            0,
524            ptr::null_mut::<libc::timespec>(),
525        ) as isize
526    })?;
527    for i in 0..(msg_count as usize) {
528        meta[i] = decode_recv(&names[i], &hdrs[i].msg_hdr, hdrs[i].msg_len as usize)?;
529    }
530    Ok(msg_count as usize)
531}

541#[cfg_attr(apple_fast, allow(dead_code))] // Unused when apple_fast is enabled
542pub(crate) fn recv_single(
543    io: SockRef<'_>,
544    bufs: &mut [IoSliceMut<'_>],
545    meta: &mut [RecvMeta],
546) -> io::Result<usize> {
547    let mut name = MaybeUninit::<libc::sockaddr_storage>::uninit();
548    let mut ctrl = cmsg::Aligned(MaybeUninit::<[u8; cmsg::LEN]>::uninit());
549    let mut hdr = unsafe { mem::zeroed::<libc::msghdr>() };
550    prepare_recv(&mut bufs[0], &mut name, &mut ctrl, &mut hdr);
551    let n = loop {
552        let n = unsafe { libc::recvmsg(io.as_raw_fd(), &mut hdr, 0) };
553
554        if hdr.msg_flags & libc::MSG_TRUNC != 0 {
555            continue;
556        }
557
558        if n >= 0 {
559            break n;
560        }
561
562        let e = io::Error::last_os_error();
563        match e.kind() {
564            // Retry receiving
565            io::ErrorKind::Interrupted => continue,
566            _ => return Err(e),
567        }
568    };
569    meta[0] = decode_recv(&name, &hdr, n as usize)?;
570    Ok(1)
571}

573#[cfg_attr(apple_fast, allow(dead_code))] // Unused when apple_fast is enabled
574fn prepare_msg(
575    transmit: &Transmit<'_>,
576    dst_addr: &socket2::SockAddr,
577    hdr: &mut libc::msghdr,
578    iov: &mut libc::iovec,
579    ctrl: &mut cmsg::Aligned<[u8; cmsg::LEN]>,
580    #[allow(unused_variables)] // only used on FreeBSD & macOS
581    encode_src_ip: bool,
582    sendmsg_einval: bool,
583) {
584    iov.iov_base = transmit.contents.as_ptr() as *const _ as *mut _;
585    iov.iov_len = transmit.contents.len();
586
587    // SAFETY: Casting the pointer to a mutable one is legal,
588    // as sendmsg is guaranteed to not alter the mutable pointer
589    // as per the POSIX spec. See the section on the sys/socket.h
590    // header for details. The type is only mutable in the first
591    // place because it is reused by recvmsg as well.
592    let name = dst_addr.as_ptr() as *mut libc::c_void;
593    let namelen = dst_addr.len();
594    hdr.msg_name = name as *mut _;
595    hdr.msg_namelen = namelen;
596    hdr.msg_iov = iov;
597    hdr.msg_iovlen = 1;
598
599    hdr.msg_control = ctrl.0.as_mut_ptr() as _;
600    hdr.msg_controllen = cmsg::LEN as _;
601    let mut encoder = unsafe { cmsg::Encoder::new(hdr) };
602    let ecn = transmit.ecn.map_or(0, |x| x as libc::c_int);
603    // True for IPv4 or IPv4-Mapped IPv6
604    let is_ipv4 = transmit.destination.is_ipv4()
605        || matches!(transmit.destination.ip(), IpAddr::V6(addr) if addr.to_ipv4_mapped().is_some());
606    if is_ipv4 {
607        if !sendmsg_einval {
608            #[cfg(not(target_os = "netbsd"))]
609            {
610                encoder.push(libc::IPPROTO_IP, libc::IP_TOS, ecn as IpTosTy);
611            }
612        }
613    } else {
614        #[cfg(not(target_os = "redox"))]
615        encoder.push(libc::IPPROTO_IPV6, libc::IPV6_TCLASS, ecn);
616    }
617
618    // On apple_fast, prepare_msg is only compiled for send_single (fallback path), while the main
619    // send path uses prepare_msg_x with msghdr_x. gso::set_segment_size has a different signature
620    // when apple_fast is enabled, and it's a no-op on non-Linux platforms anyway.
621    #[cfg(not(apple_fast))]
622    if let Some(segment_size) = transmit.effective_segment_size() {
623        gso::set_segment_size(&mut encoder, segment_size as u16);
624    }
625
626    if let Some(ip) = &transmit.src_ip {
627        match ip {
628            IpAddr::V4(v4) => {
629                #[cfg(any(target_os = "linux", target_os = "android"))]
630                {
631                    let pktinfo = libc::in_pktinfo {
632                        ipi_ifindex: 0,
633                        ipi_spec_dst: libc::in_addr {
634                            s_addr: u32::from_ne_bytes(v4.octets()),
635                        },
636                        ipi_addr: libc::in_addr { s_addr: 0 },
637                    };
638                    encoder.push(libc::IPPROTO_IP, libc::IP_PKTINFO, pktinfo);
639                }
640                #[cfg(any(bsd, apple, solarish))]
641                {
642                    if encode_src_ip {
643                        let addr = libc::in_addr {
644                            s_addr: u32::from_ne_bytes(v4.octets()),
645                        };
646                        encoder.push(libc::IPPROTO_IP, libc::IP_RECVDSTADDR, addr);
647                    }
648                }
649            }
650            #[cfg(target_os = "redox")]
651            IpAddr::V6(_) => {}
652            #[cfg(not(target_os = "redox"))]
653            IpAddr::V6(v6) => {
654                let pktinfo = libc::in6_pktinfo {
655                    ipi6_ifindex: 0,
656                    ipi6_addr: libc::in6_addr {
657                        s6_addr: v6.octets(),
658                    },
659                };
660                encoder.push(libc::IPPROTO_IPV6, libc::IPV6_PKTINFO, pktinfo);
661            }
662        }
663    }
664
665    encoder.finish();
666}

669fn prepare_recv(
670    buf: &mut IoSliceMut<'_>,
671    name: &mut MaybeUninit<libc::sockaddr_storage>,
672    ctrl: &mut cmsg::Aligned<MaybeUninit<[u8; cmsg::LEN]>>,
673    hdr: &mut libc::msghdr,
674) {
675    hdr.msg_name = name.as_mut_ptr() as _;
676    hdr.msg_namelen = mem::size_of::<libc::sockaddr_storage>() as _;
677    hdr.msg_iov = buf as *mut IoSliceMut<'_> as *mut libc::iovec;
678    hdr.msg_iovlen = 1;
679    hdr.msg_control = ctrl.0.as_mut_ptr() as _;
680    hdr.msg_controllen = cmsg::LEN as _;
681    hdr.msg_flags = 0;
682}

684pub(crate) fn decode_recv<M: cmsg::MsgHdr<ControlMessage = libc::cmsghdr>>(
685    name: &MaybeUninit<libc::sockaddr_storage>,
686    hdr: &M,
687    len: usize,
688) -> io::Result<RecvMeta> {
689    let name = unsafe { name.assume_init() };
690    let mut ctrl = ControlMetadata {
691        ecn_bits: 0,
692        dst_ip: None,
693        interface_index: None,
694        stride: len,
695    };
696
697    let cmsg_iter = unsafe { cmsg::Iter::new(hdr) };
698    for cmsg in cmsg_iter {
699        ctrl.decode(cmsg);
700    }
701
702    Ok(RecvMeta {
703        len,
704        stride: ctrl.stride,
705        addr: decode_socket_addr(&name)?,
706        ecn: EcnCodepoint::from_bits(ctrl.ecn_bits),
707        dst_ip: ctrl.dst_ip,
708        interface_index: ctrl.interface_index,
709    })
710}

712/// Metadata decoded from control messages
713struct ControlMetadata {
714    ecn_bits: u8,
715    dst_ip: Option<IpAddr>,
716    interface_index: Option<u32>,
717    stride: usize,
718}
719
720impl ControlMetadata {
721    /// Decodes a control message and updates the metadata state
722    fn decode(&mut self, cmsg: &libc::cmsghdr) {
723        match (cmsg.cmsg_level, cmsg.cmsg_type) {
724            (libc::IPPROTO_IP, libc::IP_TOS) => unsafe {
725                self.ecn_bits = cmsg::decode::<u8, libc::cmsghdr>(cmsg);
726            },
727            // FreeBSD uses IP_RECVTOS here, and we can be liberal because cmsgs are opt-in.
728            #[cfg(not(any(
729                target_os = "openbsd",
730                target_os = "netbsd",
731                target_os = "dragonfly",
732                solarish
733            )))]
734            (libc::IPPROTO_IP, libc::IP_RECVTOS) => unsafe {
735                self.ecn_bits = cmsg::decode::<u8, libc::cmsghdr>(cmsg);
736            },
737            #[cfg(not(target_os = "redox",))]
738            (libc::IPPROTO_IPV6, libc::IPV6_TCLASS) => unsafe {
739                // Temporary hack around broken macos ABI. Remove once upstream fixes it.
740                // https://bugreport.apple.com/web/?problemID=48761855
741                #[allow(clippy::unnecessary_cast)] // cmsg.cmsg_len defined as size_t
742                if cfg!(apple)
743                    && cmsg.cmsg_len as usize == libc::CMSG_LEN(mem::size_of::<u8>() as _) as usize
744                {
745                    self.ecn_bits = cmsg::decode::<u8, libc::cmsghdr>(cmsg);
746                } else {
747                    self.ecn_bits = cmsg::decode::<libc::c_int, libc::cmsghdr>(cmsg) as u8;
748                }
749            },
750            #[cfg(any(target_os = "linux", target_os = "android"))]
751            (libc::IPPROTO_IP, libc::IP_PKTINFO) => {
752                let pktinfo = unsafe { cmsg::decode::<libc::in_pktinfo, libc::cmsghdr>(cmsg) };
753                self.dst_ip = Some(IpAddr::V4(Ipv4Addr::from(
754                    pktinfo.ipi_addr.s_addr.to_ne_bytes(),
755                )));
756                self.interface_index = Some(pktinfo.ipi_ifindex as u32);
757            }
758            #[cfg(any(bsd, apple))]
759            (libc::IPPROTO_IP, libc::IP_RECVDSTADDR) => {
760                let in_addr = unsafe { cmsg::decode::<libc::in_addr, libc::cmsghdr>(cmsg) };
761                self.dst_ip = Some(IpAddr::V4(Ipv4Addr::from(in_addr.s_addr.to_ne_bytes())));
762            }
763            #[cfg(not(target_os = "redox",))]
764            (libc::IPPROTO_IPV6, libc::IPV6_PKTINFO) => {
765                let pktinfo = unsafe { cmsg::decode::<libc::in6_pktinfo, libc::cmsghdr>(cmsg) };
766                self.dst_ip = Some(IpAddr::V6(Ipv6Addr::from(pktinfo.ipi6_addr.s6_addr)));
767                #[cfg_attr(not(target_os = "android"), expect(clippy::unnecessary_cast))]
768                {
769                    self.interface_index = Some(pktinfo.ipi6_ifindex as u32);
770                }
771            }
772            #[cfg(any(target_os = "linux", target_os = "android"))]
773            (libc::SOL_UDP, libc::UDP_GRO) => unsafe {
774                self.stride = cmsg::decode::<libc::c_int, libc::cmsghdr>(cmsg) as usize;
775            },
776            _ => {}
777        }
778    }
779}

781/// Decodes a `sockaddr_storage` into a `SocketAddr`
782pub(crate) fn decode_socket_addr(name: &libc::sockaddr_storage) -> io::Result<SocketAddr> {
783    match libc::c_int::from(name.ss_family) {
784        libc::AF_INET => {
785            // Safety: if the ss_family field is AF_INET then storage must be a sockaddr_in.
786            let addr: &libc::sockaddr_in =
787                unsafe { &*(name as *const _ as *const libc::sockaddr_in) };
788            Ok(SocketAddr::V4(SocketAddrV4::new(
789                Ipv4Addr::from(addr.sin_addr.s_addr.to_ne_bytes()),
790                u16::from_be(addr.sin_port),
791            )))
792        }
793        libc::AF_INET6 => {
794            // Safety: if the ss_family field is AF_INET6 then storage must be a sockaddr_in6.
795            let addr: &libc::sockaddr_in6 =
796                unsafe { &*(name as *const _ as *const libc::sockaddr_in6) };
797            Ok(SocketAddr::V6(SocketAddrV6::new(
798                Ipv6Addr::from(addr.sin6_addr.s6_addr),
799                u16::from_be(addr.sin6_port),
800                addr.sin6_flowinfo,
801                addr.sin6_scope_id,
802            )))
803        }
804        f => Err(io::Error::other(format!(
805            "expected AF_INET or AF_INET6, got {f}"
806        ))),
807    }
808}

844/// Returns whether the given socket option is supported on the current platform
845///
846/// Yields `Ok(true)` if the option was set successfully, `Ok(false)` if setting
847/// the option raised an `ENOPROTOOPT` or `EOPNOTSUPP` error, and `Err` for any other error.
848fn set_socket_option_supported(
849    socket: &impl AsRawFd,
850    level: libc::c_int,
851    name: libc::c_int,
852    value: libc::c_int,
853) -> io::Result<bool> {
854    match set_socket_option(socket, level, name, value) {
855        Ok(()) => Ok(true),
856        Err(err) if err.raw_os_error() == Some(libc::ENOPROTOOPT) => Ok(false),
857        Err(err) if err.raw_os_error() == Some(libc::EOPNOTSUPP) => Ok(false),
858        Err(err) => Err(err),
859    }
860}
861
862pub(crate) fn set_socket_option(
863    socket: &impl AsRawFd,
864    level: libc::c_int,
865    name: libc::c_int,
866    value: libc::c_int,
867) -> io::Result<()> {
868    let rc = unsafe {
869        libc::setsockopt(
870            socket.as_raw_fd(),
871            level,
872            name,
873            &value as *const _ as _,
874            mem::size_of_val(&value) as _,
875        )
876    };
877
878    match rc == 0 {
879        true => Ok(()),
880        false => Err(io::Error::last_os_error()),
881    }
882}

886/// Calls `f` in a loop, retrying on `EINTR`.
887///
888/// Returns the non-negative result or the first non-`EINTR` error.
889pub(crate) fn retry_if_interrupted(mut f: impl FnMut() -> isize) -> io::Result<isize> {
890    loop {
891        let n = f();
892        if n >= 0 {
893            return Ok(n);
894        }
895        let e = io::Error::last_os_error();
896        if e.kind() != io::ErrorKind::Interrupted {
897            return Err(e);
898        }
899    }
